Algorithmique visuelle

carre.py
Logique de comparaison

Script de recherche de valeur minimale entre deux nombres réels flottants.

minimum.py
Code : ip.py
ip = input("Adresse ip ? ").split(".")
if int(ip[3]) == 0:
    print("Adresse réseau")
elif int(ip[3]) == 255:
    print("Adresse broadcast")
else:
    print("Adresse machine")
Code : liste.py
n = int(input("n ?"))
l = []
for i in range (n):
    l.append(int(input(f"l[{i}]= ")))
print(f"Liste créer : {l}")
lm = sum(l)
lm2 = lm / n
print(f"La moyenne est de {round (lm2, 2)}")
Code : carre.py
n = int(input("n ? "))
for i in range (1):
    print("*"*n)
    for i in range(n-2):
        print("*","*", sep=" "*int(n-2))
    print("*"*n)
Code : minimum.py
r1 = float(input("Nombre 1 : "))
r2 = float(input("Nombre 2 : "))
min_val = min(r1, r2)
print(f"Le minimum est {min_val}")
